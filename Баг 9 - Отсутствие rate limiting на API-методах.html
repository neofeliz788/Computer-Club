Issue #9

УГРОЗА БЕЗОПАСНОСТИ: DDoS АТАКА

Отсутствие rate limiting делает систему уязвимой к DDoS-атакам. Злоумышленник может легко исчерпать ресурсы системы, отправляя большое количество запросов к критическим endpoint'ам.

Описание

Критические API-методы (авторизация, создание брони) не имеют ограничений по частоте запросов. Это позволяет одному клиенту отправлять неограниченное количество запросов в секунду, что может привести к отказу в обслуживании для легитимных пользователей.

Шаги воспроизведения

  1. Написать скрипт на Python/Node.js/Bash, который отправляет 1000 запросов в секунду на endpoint авторизации (/api/auth/login).
  2. Запустить скрипт на 1 минуту (60 000 запросов).
  3. Мониторить метрики системы во время атаки:
    • Загрузка CPU сервера
    • Использование памяти
    • Количество активных соединений к БД
    • Время ответа для легитимных пользователей
  4. Попытаться использовать систему как обычный пользователь во время атаки.
  5. Повторить тест для endpoint'а создания брони (/api/bookings).

Последствия атаки

Высокая нагрузка на CPU

Сервер пытается обработать все 1000 запросов в секунду, что приводит к 100% загрузке CPU. Это замедляет обработку всех запросов, включая легитимные.

Исчерпание памяти

Каждый запрос создает объекты в памяти (сессии, соединения с БД, объекты запросов). При высокой нагрузке память быстро заполняется, что может привести к падению приложения.

Отказ в обслуживании

Легитимные пользователи не могут использовать систему из-за исчерпания ресурсов. Время ответа возрастает с миллисекунд до десятков секунд.

Ожидаемый результат

Ожидаемый результат

После 10-20 запросов в секунду система должна возвращать HTTP статус 429 Too Many Requests. Легитимные пользователи должны иметь возможность работать с системой даже во время атаки.

Фактический результат

Фактический результат

Система пытается обработать все запросы, что приводит к:

  • Высокой нагрузке на CPU (до 100%)
  • Исчерпанию памяти
  • Отказу в обслуживании легитимных пользователей
  • Возможности DDoS-атаки

Решение 1: ASP.NET Core Middleware с IMemoryCache

public class RateLimitMiddleware {
    private readonly RequestDelegate _next;
    private readonly IMemoryCache _cache;
    
    public RateLimitMiddleware(RequestDelegate next, IMemoryCache cache) {
        _next = next;
        _cache = cache;
    }
    
    public async Task InvokeAsync(HttpContext context) {
        var endpoint = context.Request.Path;
        var clientIp = context.Connection.RemoteIpAddress?.ToString() ?? "unknown";
        var clientKey = $"{clientIp}:{endpoint}";
        
        // Получаем или создаем счетчик запросов
        var requestCounter = _cache.GetOrCreate(clientKey, entry => {
            entry.AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(1);
            return new RequestCounter();
        });
        
        // Проверяем лимит
        if (requestCounter.Count >= GetRateLimit(endpoint)) {
            context.Response.StatusCode = 429; // Too Many Requests
            context.Response.Headers.Add("Retry-After", "60");
            await context.Response.WriteAsync("Rate limit exceeded. Please try again later.");
            return;
        }
        
        // Увеличиваем счетчик
        requestCounter.Count++;
        _cache.Set(clientKey, requestCounter);
        
        await _next(context);
    }
    
    private int GetRateLimit(string endpoint) {
        return endpoint switch {
            string s when s.Contains("/auth/") => 10,
            string s when s.Contains("/bookings") && context.Request.Method == "POST" => 5,
            _ => 100
        };
    }
    
    private class RequestCounter {
        public int Count { get; set; }
    }
}

// В Startup.cs
app.UseMiddleware<RateLimitMiddleware>();

            Endpoint = "*:/api/auth/*",

            Endpoint = "POST:/api/bookings",
            Period = "1m",
            Limit = 5

        ContentType = "application/json",

    };

Решение 3: Распределенный rate limiting с Redis

public class DistributedRateLimiter {
    private readonly IConnectionMultiplexer _redis;
    
    public DistributedRateLimiter(IConnectionMultiplexer redis) {
        _redis = redis;
    }
    
    public async Task<bool> IsAllowedAsync(string key, int limit, TimeSpan period) {
        var db = _redis.GetDatabase();
        var currentTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        var windowKey = $"{key}:{currentTime / (int)period.TotalSeconds}";
        
        // Используем Redis INCR с TTL
        var current = await db.StringIncrementAsync(windowKey);
        if (current == 1) {
            await db.KeyExpireAsync(windowKey, period);
        }
        
        return current <= limit;
    }
}

// Использование
public class RateLimitedController : ControllerBase {
    private readonly DistributedRateLimiter _rateLimiter;
    
    [HttpPost("auth/login")]
    public async Task<IActionResult> Login(LoginRequest request) {
        var clientIp = HttpContext.Connection.RemoteIpAddress?.ToString();
        var key = $"rate_limit:{clientIp}:login";
        
        if (!await _rateLimiter.IsAllowedAsync(key, 10, TimeSpan.FromMinutes(1))) {
            Response.Headers.Add("Retry-After", "60");
            return StatusCode(429, new { error = "Too many login attempts" });
        }
        
        // Логика аутентификации
        return Ok(new { token = "..." });
    }
}

Решение 4: WAF или API Gateway уровень

# Пример конфигурации NGINX как API Gateway
http {
    limit_req_zone $binary_remote_addr zone=auth:10m rate=10r/m;
    limit_req_zone $binary_remote_addr zone=bookings:10m rate=5r/m;
    
    server {
        listen 80;
        
        location /api/auth/ {
            limit_req zone=auth burst=5 nodelay;
            limit_req_status 429;
            proxy_pass http://backend:5000;
        }
        
        location /api/bookings {
            limit_req zone=bookings burst=3 nodelay;
            limit_req_status 429;
            proxy_pass http://backend:5000;
        }
        
        location / {
            proxy_pass http://backend:5000;
        }
    }
}

Решение 5: Адаптивный rate limiting

public class AdaptiveRateLimiter {
    private readonly ConcurrentDictionary<string, ClientStats> _stats = new();
    
    public async Task<RateLimitResult> CheckLimitAsync(string clientId, string endpoint) {
        var stats = _stats.GetOrAdd($"{clientId}:{endpoint}", 
            key => new ClientStats());
        
        // Анализ паттерна запросов
        if (stats.IsSuspiciousPattern()) {
            // Ужегительный лимит для подозрительных клиентов
            return new RateLimitResult {
                IsAllowed = stats.RequestsLastMinute < 2,
                RetryAfterSeconds = 120
            };
        }
        
        // Нормальный лимит
        var limit = GetDefaultLimit(endpoint);
        return new RateLimitResult {
            IsAllowed = stats.RequestsLastMinute < limit,
            RetryAfterSeconds = 60
        };
    }
    
    private bool IsSuspiciousPattern(ClientStats stats) {
        // Обнаружение паттернов атаки
        return stats.RequestsLastSecond > 50 || 
               stats.ConsecutiveFailures > 10 ||
               stats.TotalRequestsToday > 1000;
    }
}

// Настройка алертов при превышении лимитов
public class RateLimitAlertService : BackgroundService {
    protected override async Task ExecuteAsync(CancellationToken stoppingToken) {
        while (!stoppingToken.IsCancellationRequested) {
            var exceededLimits = await GetExceededRateLimitsAsync();
            if (exceededLimits.Any()) {
                await SendAlertAsync($"Rate limits exceeded: {string.Join(", ", exceededLimits)}");
            }
            await Task.Delay(TimeSpan.FromMinutes(1), stoppingToken);
        }
    }
}
